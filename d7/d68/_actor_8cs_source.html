314 protected void RotateActor(RotateType RotateDir)
315 {
316 int tempDir = (int)_moveDirection - 1;
317 if (RotateDir == RotateType.Random)
318 RotateDir = (RotateType)Random.Range(1, 5);
319 switch (RotateDir)
320 {
321 case RotateType.Front:
322 break;
323 case RotateType.Back:
324 tempDir += 2;
325 tempDir %= 4;
326 break;
327 case RotateType.Left:
328 tempDir--;
329 if (tempDir < 0)
330 {
331 tempDir += 4;
332 }
333 break;
334 case RotateType.Right:
335 tempDir++;
336 tempDir %= 4;
337 break;
338 case RotateType.TargetPlayer:
339 tempDir = (int)_cellRouteTrackingSystem.GetRouteTrackingDir(_cellBasedObject.CurrentCellPosition, _actorSystem.Player.GetCellPosition()) - 1;
340 break;
341 }
342 _moveDirection = (CellDirection)(tempDir + 1);
343 _currentDirection = _moveDirection;
344 }
