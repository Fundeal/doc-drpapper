151 private void ManaHauguleCombination(string consonant)
152 {
153 _hanguleList.Add(consonant);
154 if (_hanguleList.Count >= 4)
155 {
156 string combiTemp = HangulUtil.Combination(_hanguleList.GetRange(2,2));
157 if (!HangulDecomposition.IsLastJaum(combiTemp))
158 {
159 _hanguleList.Clear();
160 _hanguleList.Add(consonant);
161 manaText.text = _hanguleList[0];
162 return;
163 }
164 }
165 else if (_hanguleList.Count >= 3)
166 {
167 if (!HangulDecomposition.IsLastJaum(_hanguleList[2]))
168 {
169 _hanguleList.Clear();
170 _hanguleList.Add(consonant);
171 manaText.text = _hanguleList[0];
172 return;
173 }
174 }
175 else if (_hanguleList.Count >= 2)
176 {
177 if (!HangulDecomposition.IsMoum(_hanguleList[1]))
178 {
179 _hanguleList.Clear();
180 _hanguleList.Add(consonant);
181 manaText.text = _hanguleList[0];
182 return;
183 }
184 else if (!HangulDecomposition.IsJaum(_hanguleList[0]))
185 {
186 _hanguleList.Clear();
187 _hanguleList.Add(consonant);
188 manaText.text = _hanguleList[0];
189 return;
190 }
191 }
192 else
193 {
194 manaText.text = _hanguleList[0];
195 return;
196 }
197 manaText.text = HangulUtil.Combination(_hanguleList);
198 EnemyHanguleDeath();
199 IsAttack = true;
200 }
