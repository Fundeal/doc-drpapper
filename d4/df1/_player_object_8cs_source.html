204 private void ManaHauguleCombination(string consonant)
205 {
206 _hanguleList.Add(consonant);
207 if (_hanguleList.Count >= 4)
208 {
209 string combiTemp = HangulUtil.Combination(_hanguleList.GetRange(2,2));
210 if (!HangulDecomposition.IsLastJaum(combiTemp))
211 {
212 _hanguleList.Clear();
213 _hanguleList.Add(consonant);
214 manaText.text = _hanguleList[0];
215 return;
216 }
217 }
218 else if (_hanguleList.Count >= 3)
219 {
220 if (!HangulDecomposition.IsLastJaum(_hanguleList[2]))
221 {
222 _hanguleList.Clear();
223 _hanguleList.Add(consonant);
224 manaText.text = _hanguleList[0];
225 return;
226 }
227 }
228 else if (_hanguleList.Count >= 2)
229 {
230 if (!HangulDecomposition.IsMoum(_hanguleList[1]))
231 {
232 _hanguleList.Clear();
233 _hanguleList.Add(consonant);
234 manaText.text = _hanguleList[0];
235 return;
236 }
237 else if (!HangulDecomposition.IsJaum(_hanguleList[0]))
238 {
239 _hanguleList.Clear();
240 _hanguleList.Add(consonant);
241 manaText.text = _hanguleList[0];
242 return;
243 }
244 }
245 else
246 {
247 manaText.text = _hanguleList[0];
248 return;
249 }
250 string combinedText = HangulUtil.Combination(_hanguleList);
251 manaText.text = combinedText;
252
253 if(SystemManager.Instance.GetSystem<Core.Room.DungeonSystem>().UpdateGoal(combinedText))
254 {
255 Debug.Log("단어 완성함");
256 }
257
258 EnemyHanguleDeath();
259 IsAttack = true;
